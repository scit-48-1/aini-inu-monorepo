   ```
   const handleEdit = () => {
     setEditContent(post.content);
     setIsEditing(true);
     setShowMenu(false);
   };

   const handleSaveEdit = async () => {
     if (!editContent.trim()) {
       toast.error('내용을 입력해주세요.');
       return;
     }
     setIsSaving(true);
     try {
       const updatedPost = await updatePost(post.id, { content: editContent });
       setPost(prev => ({ ...prev, content: updatedPost.content }));
       toast.success('게시글이 수정되었습니다.');
       onEditUpdate?.(updatedPost);
       setIsEditing(false);
     } catch {
       toast.error('수정에 실패했습니다.');
     } finally {
       setIsSaving(false);
     }
   };

   const handleCancelEdit = () => {
     setIsEditing(false);
     setEditContent('');
   };
   ```

5. Replace the ENTIRE dropdown edit button conditional (lines 304-317) with a single button that calls handleEdit directly:

   ```
   <button
     onClick={(e) => { e.stopPropagation(); handleEdit(); }}

   >
     <Pencil size={14} /> 수정
   </button>
   ```
   This removes both branches (the onEdit callback branch AND the toast stub branch).

   ```
   {/* Content */}

     {isEditing ? (
       <div className="space-y-3" onClick={(e) => e.stopPropagation()}>
         <textarea
           className="w-full bg-zinc-50 border border-zinc-100 rounded-2xl p-4 text-sm font-medium focus:outline-none focus:ring-2 ring-amber-500/20 min-h-[100px] resize-none"
           value={editContent}
           onChange={(e) => setEditContent(e.target.value)}
           placeholder="글 내용을 입력해주세요..."

             {isSaving ? '저장 중...' : '저장'}

   </div>
   ```
